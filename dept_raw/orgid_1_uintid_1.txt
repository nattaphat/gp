	function clickClear() {
		var f1 = document.form1;
		f1.ORG_ID.disabled=false;
		f1.ORG_UNIT_ID.disabled=false;
		f1.QUERY_TYPE.value="";
		f1.QUERY.value="";
		f1.PROV_ID.value="";

		f1.QUERY_TYPE[1].checked=false;

	function clickSearch() {
		var f1 = document.form1;
		
		if (f1.QUERY_TYPE[0].checked || f1.QUERY_TYPE[1].checked) {
			if (f1.QUERY.value.length < 4) {
				alert(showMessage(E1001)+"รหัส/ชื่อหน่วยงาน อย่างน้อย 4 ตัวอักษร");
				f1.QUERY.focus();
				return false;
			}
		} else {
			if (f1.QUERY.value != "") {
				alert(showMessage(E1001)+" ค้นหาจาก รหัสหรือชื่อหน่วยงาน");
				f1.QUERY_TYPE[0].focus();
				return false;
			}
		}
		
		f1.ORG_ID.disabled=false;
		f1.ORG_UNIT_ID.disabled=false;

		f1.flag.value="SEARCH";
//		action="./FAGN0103H.jsp";
		f1.action="control.egp";
		f1.submit();
